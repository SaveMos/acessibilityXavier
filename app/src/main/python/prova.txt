const val FS = 500

import kotlin.math.*

fun butterworthLowpass(data: DoubleArray, cutoff: Double, fs: Int, order: Int = 4): DoubleArray {
    val nyq = 0.5 * fs
    val normCutoff = cutoff / nyq

    val (b, a) = butterCoefficientsLowpass(normCutoff, order)
    return lfilter(b, a, data)
}

fun butterworthBandpass(data: DoubleArray, lowcut: Double, highcut: Double, fs: Int, order: Int = 4): DoubleArray {
    val nyq = 0.5 * fs
    val low = lowcut / nyq
    val high = highcut / nyq

    val (b, a) = butterCoefficientsBandpass(low, high, order)
    return lfilter(b, a, data)
}

fun butterCoefficientsLowpass(cutoff: Double, order: Int): Pair<DoubleArray, DoubleArray> {
    val theta = PI * cutoff
    val d = DoubleArray(2 * order + 1)
    val a = DoubleArray(order + 1)
    val b = DoubleArray(order + 1)

    val gain = tan(theta)
    val gSquared = gain * gain
    val denom = 1.0 + sqrt(2.0) * gain + gSquared

    b[0] = gSquared / denom
    b[1] = 2 * b[0]
    b[2] = b[0]

    a[0] = 1.0
    a[1] = 2.0 * (gSquared - 1.0) / denom
    a[2] = (1.0 - sqrt(2.0) * gain + gSquared) / denom

    return Pair(b, a)
}

// Simplified 2nd order Butterworth bandpass for illustration
fun butterCoefficientsBandpass(low: Double, high: Double, order: Int): Pair<DoubleArray, DoubleArray> {
    val center = sqrt(low * high)
    val bandwidth = high - low

    // This is an approximation; a full bandpass Butterworth requires cascade of lowpass and highpass sections.
    val theta = PI * bandwidth
    val gain = tan(theta / 2.0)
    val gSquared = gain * gain
    val normCenter = 2 * PI * center

    val denom = 1 + gain
    val b = doubleArrayOf(gain / denom, 0.0, -gain / denom)
    val a = doubleArrayOf(1.0, 0.0, (1 - gain) / denom)

    return Pair(b, a)
}

// Apply filter (direct-form I)
fun lfilter(b: DoubleArray, a: DoubleArray, data: DoubleArray): DoubleArray {
    val y = DoubleArray(data.size)
    for (i in data.indices) {
        y[i] = b[0] * data.getOrElse(i) { 0.0 } +
               (if (i >= 1) b.getOrElse(1) * data[i - 1] else 0.0) +
               (if (i >= 2) b.getOrElse(2) * data[i - 2] else 0.0) -
               (if (i >= 1) a.getOrElse(1) * y[i - 1] else 0.0) -
               (if (i >= 2) a.getOrElse(2) * y[i - 2] else 0.0)
    }
    return y
}

fun combineFilteredChannels(data: Array<DoubleArray>, method: String = "sum"): DoubleArray {
    val nSamples = data.size
    val nChannels = data[0].size
    val combined = DoubleArray(nSamples) { i ->
        when (method) {
            "mean" -> data[i].average()
            "max" -> data[i].maxOrNull() ?: 0.0
            else -> data[i].sum()
        }
    }

    val lowpassed = butterworthLowpassFilter(combined, 10.0, FS, 4)
    return butterworthBandpassFilter(lowpassed, 10.0, 30.0, FS, 4)
}

fun computeWelch(signal: DoubleArray, fs: Int): Pair<DoubleArray, DoubleArray> {
    val n = signal.size
    val fft = DoubleFFT_1D(n.toLong())
    val padded = signal.copyOf(n * 2)
    fft.realForward(padded)
    val psd = DoubleArray(n / 2) { i ->
        val re = padded[2 * i]
        val im = padded[2 * i + 1]
        (re * re + im * im) / n
    }
    val freqs = DoubleArray(psd.size) { i -> i.toDouble() * fs / n }
    return Pair(freqs, psd)
}

fun bandPower(freqs: DoubleArray, psd: DoubleArray, low: Double, high: Double): Double {
    val indices = freqs.indices.filter { freqs[it] in low..high }
    val freqSlice = indices.map { freqs[it] }.toDoubleArray()
    val psdSlice = indices.map { psd[it] }.toDoubleArray()
    return trapz(psdSlice, freqSlice)
}

fun trapz(y: DoubleArray, x: DoubleArray): Double {
    var sum = 0.0
    for (i in 0 until y.size - 1) {
        val dx = x[i + 1] - x[i]
        val dy = (y[i] + y[i + 1]) / 2.0
        sum += dx * dy
    }
    return sum
}

// Load model and scaler
lateinit var model: Classifier<DoubleArray>
lateinit var scaler: Standardizer

fun loadModelAndScaler(modelPath: String, scalerPath: String) {
    ObjectInputStream(FileInputStream(File(modelPath))).use {
        model = it.readObject() as Classifier<DoubleArray>
    }
    ObjectInputStream(FileInputStream(File(scalerPath))).use {
        scaler = it.readObject() as Standardizer
    }
}

fun analyzeLiveBlink(buffer: ByteArray, nChannels: Int): Int {
    return try {
        val floatBuffer = buffer.asFloatBuffer()
        val floatArray = FloatArray(floatBuffer.remaining())
        floatBuffer.get(floatArray)

        val trimmed = floatArray.map { it.toDouble() }.toDoubleArray()
        val totalSamples = trimmed.size / nChannels
        val reshaped = Array(totalSamples) { row ->
            DoubleArray(nChannels) { col ->
                trimmed[row * nChannels + col]
            }
        }

        val combinedSignal = combineFilteredChannels(reshaped)
        val (freqs, psd) = computeWelch(combinedSignal, FS)

        val alpha = bandPower(freqs, psd, 8.0, 12.9)
        val beta = bandPower(freqs, psd, 13.0, 29.9)
        val gamma = bandPower(freqs, psd, 30.0, 50.0)
        val theta = bandPower(freqs, psd, 4.0, 7.9)

        val input = doubleArrayOf(alpha, beta, gamma, theta)
        val scaled = scaler.transform(input)

        val prediction = model.predict(scaled)
        prediction
    } catch (e: Exception) {
        e.printStackTrace()
        0
    }
}
